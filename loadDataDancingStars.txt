CREATE or replace PROCEDURE P_GEN_DATE()
BEGIN
    DECLARE I INT;
    SET I = 1;
    WHILE I <= 149
        DO
            INSERT INTO JURY_SCORE(PERFORMANCE_ID, JURY_ID, SCORE)
            VALUES (I, 1, 5 + MOD(MICROSECOND(CURRENT_TIMESTAMP), 6)),
                   (I, 2, 5 + MOD(MICROSECOND(CURRENT_TIMESTAMP), 6)),
                   (I, 3, 5 + MOD(MICROSECOND(CURRENT_TIMESTAMP), 6)),
                   (I, 4, 5 + MOD(MICROSECOND(CURRENT_TIMESTAMP), 6));
            SET I = I + 1;
        END WHILE;
END;
CALL FN24_5MI0700203.P_GEN_DATE();
